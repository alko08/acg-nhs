Import all the stuff from default libraries and such


public class teleop {
    public static void init() {
        Do some stuff with declaration of motors and stuff
        loop()
    }

    public static void loop() {
        if Right Trigger is pressed and held
            go forward
        if Left Trigger is pressed and held
            go backwards
            
            if Right Bumper is also pressed
                turn right
            if Left Bumper is also pressed
                turn left
        if B Button is pressed
            emergency stop
        if both triggers are pressed
            do nothing
            
        loop every few milliseconds
        
    
    }





}
